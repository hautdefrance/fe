<style lang="stylus" scoped>
  .wrapper {

# N栏布局(Grid布局)

Source
<template>
  <div class="css-flexbox wrapper">
    <div v-for="(item, i) in list" :key="i" class="row">
      <div v-for="(subitem, j) in item" :key="j" class="item"></div>
    </div>
  </div>
</template>

<script>
  function createSparseArray (length) {
    return Object.keys(Array.apply(null, { length }))
  }
  export default {
    data() {
      return {
        list: createSparseArray(5).map((item, index) => createSparseArray(index + 1))
      }
    }
  }
</script>

<style lang="stylus" scoped>
  $gutter = 5px

  .css-flexbox {
    padding: $gutter;
    background-color: #eee;
  }

  .row {
    display: flex;
    margin-bottom $gutter;
    &:last-child {
      margin-bottom 0px;
    }
  }

  .item {
    flex: 1;
    height: 30px;
    background-color: #fff;
    margin-right: $gutter;
    &:last-child {
      margin-right: 0px;
    }
  }
</style>

# 圣杯布局(Holy Grail Layout)

Header
Content
Nav
Aside
Footer
Source
<template>
  <div class="wrapper shadow-card">
    <header class="hg-header"> Header </header>
    <div class="hg-container">
      <main class="hg-content">Content</main>
      <nav class="hg-nav">Nav</nav>
      <aside class="hg-aside">Aside</aside>
    </div>
    <footer class="hg-footer"> Footer </footer>
  </div>
</template>

<style lang="stylus" scoped>
  .wrapper {
    display: flex;
    flex-direction: column;
    @extend $shadow;
    .hg-header, .hg-footer {
      text-align: center;
      height: 30px;
      line-height: 30px;
      color: $gray;
      background-color: $light;
    }
    .hg-container {
      display: flex;
      .hg-content, .hg-nav, .hg-aside {
        line-height: 100px;
        text-align: center;
        color: white;
      }
      .hg-nav, .hg-aside {
        flex: 0 0 100px;
        background-color $red;
        height: 100px;
      }
      .hg-content {
        flex: 1;
        background-color $green;
        height: 100px;
      }

      .hg-nav {
        order: -1;
      }
    }
  }
</style>
